        visiting.remove(skill);
        visited.add(skill);
        order.add(skill);

    }

            stat.setResourceName(resource);
            stat.setCurrentUsage(currentUsage);
            stat.setMaxUsage(maxUsage);
            stat.setUtilizationRate((double) currentUsage / maxUsage);

            stats.put(resource, stat);
        }

        return stats;
    }
